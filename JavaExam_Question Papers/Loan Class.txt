public class Loan{
    private double rate;
    private int period;
    private double loanAmount;
    private java.util.Data loanDate;

    // Default(Empty) constructor
    public Loan(){
        this(2.5, 1, 1000);
    }

    // Parameterized Constructor
    public Loan(double _rate, int _period, double _loanAmount){
        this.rate = _rate;
        this.period = _period;
        this.loanAmount = _loanAmount;
        loanDate = new java.util.Date();
    }

    // Return AnnualInterestRate
    public double getInterestRate(){
        return this.rate;
    }

    // Set a new AnnualInterestRate
    public void setInterestRate(double _rate){
        this.rate = _rate;
    }

    // Return period
    public int getPeriod(){
        return period;
    }

    // Set period
    public void setPeriod(int _period){
        this.period = _period;
    }

    // Return loanAmount   
    public double getLoanAmount(){
        return loanAmount;
    }

    // Set a new loanAmount
    public void setLoanAmount(double _loanAmount){
        this.loanAmount = _loanAmount;
    }

    // Calculate monthly payment
    public double calculateEMI(){
        double monthlyRate = rate / 1200;
        double monthlyPayment = loanAmount * monthlyRate / (1 - (1 / Math.pow(1 + monthlyRate, period * 12)));
        return monthlyPayment;
    }

    // Calculate total payment
    public double getTotalPayment(){
        double totalPayment = calculateEMI * period * 12;
        return totalPayment;
    }

    // Return loan Date
    public java.util.Date getLoanDate(){
        return loanDate;
    }



}
