abstract class PlainDataObject{
    public int idNumber = 99;
    public String dataName = "No name";
    
    public PlainDataObject() {  }
    
    public PlainDataObject(int id, String name) {
        idNumber = id;
        dataName = name;
    }
    
    public void print(){
        System.out.printf("DataObject : id : %3d", idNumber);
        System.out.print(",name: " + dataName);
        return;
    }
}

class ExtendedDataObject extends PlainDataObject{
    public int value;
    
    public ExtendedDataObject(){
        super(33, "extendedDO");
        value = 44;
    }
    
    public ExtendedDataObject(int num, String name){
        dataName = name;
        value = num;
    }
    
    public void print(){
        super.print();
        System.out.print(", value: " + value);
    }
}

class AltDataObject extends PlainDataObject{
    protected int value = 10;
    
    AltDataObject(){}
    
    AltDataObject(String name){
        dataName = name;
        value = 333;
    }
    
    public void print(){
        super.print();
        int tmp = value + idNumber;
        System.out.print(" , tmp: "+ tmp);
    }
}

class LastDataObject extends AltDataObject{
    private String dataMsg;
    protected LastDataObject(){}
    
    LastDataObject(String msg){
        super("Anne");
        dataMsg = msg;
    }
    
    public void print(){
        super.print();
        System.out.print(" LastDataObject " + dataMsg);
    }
    
}

public class MainMethod
{
    public static void main(String[] args){
        PlainDataObject myObject = new AltDataObject();
        myObject.print();
        System.out.println();
        
        myObject = new ExtendedDataObject();
        myObject.print();
        System.out.println();
        
        myObject = new ExtendedDataObject(99, "Bob");
        myObject.print();
        System.out.println();
        
        AltDataObject ado = new LastDataObject(" bye bye .....");
        ado.print();
        System.out.println();
        
    }
}
